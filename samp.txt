# 3. Make the first commit

# 4. Verify you now have a commit
git log --oneline --decorate --graph --all

git branch

# 6. Switch to dev
git checkout dev

# 7. Create user-story branch from dev
git checkout -b user-story-1

git checkout -b user-story-1-task-1
# (make changes, then)
git add .

git checkout user-story-1

git add .
